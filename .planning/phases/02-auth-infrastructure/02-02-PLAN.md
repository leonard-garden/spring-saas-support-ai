    4. Implement `frontend/src/hooks/useAuthInit.ts`:
       ```ts
       import { useEffect } from "react"
       import axios from "axios"
       import { useAuthStore } from "@/store/authStore"
       import { getRefreshToken, setRefreshToken, clearRefreshToken } from "@/lib/tokenStorage"
       import type { ApiResponse, AuthResponse, MeResponse } from "@/types/auth"

       const baseURL = import.meta.env.VITE_API_URL

       export function useAuthInit() {
         const setAuth = useAuthStore((s) => s.setAuth)
         const clearAuth = useAuthStore((s) => s.clearAuth)

         useEffect(() => {
           const refreshToken = getRefreshToken()
           if (!refreshToken) {
             clearAuth()
             return
           }

           const run = async () => {
             try {
               // Use raw axios (NOT the api instance) to avoid interceptor recursion on hydration
               const refreshRes = await axios.post<ApiResponse<AuthResponse>>(
                 `${baseURL}/api/v1/auth/refresh`,
                 { refreshToken },
                 { headers: { "Content-Type": "application/json" } }
               )
               if (!refreshRes.data.success || !refreshRes.data.data) {
                 throw new Error(refreshRes.data.error ?? "Refresh failed")
               }
               const { accessToken, refreshToken: newRefreshToken } = refreshRes.data.data
               setRefreshToken(newRefreshToken)

               const meRes = await axios.get<ApiResponse<MeResponse>>(
                 `${baseURL}/api/v1/auth/me`,
                 { headers: { Authorization: `Bearer ${accessToken}` } }
               )
               if (!meRes.data.success || !meRes.data.data) {
                 throw new Error(meRes.data.error ?? "Me lookup failed")
               }
               setAuth(accessToken, meRes.data.data)
             } catch {
               clearRefreshToken()
               clearAuth()
             }
           }
           void run()
         }, [setAuth, clearAuth])
       }
       ```
